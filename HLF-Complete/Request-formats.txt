For running the Hyperledger fabric network project:<br>
./start-network.sh
<br>

To run the SDK, follow these steps: <br>
1.Build the project using IntelliJ or any other IDE. <br>
2.Generate the jar file. <br>
3.Run the jar file using the following command (to run as a background service): <br>
nohup java -jar HLF-Java-SDK-demo-0.0.1-SNAPSHOT.jar -Dspring.config.location=application.properties -Dserver.port=9000 & <br>
<br>
Request Formats <br>
You can utilize Postman or a similar tool to dispatch the request  <br>

//Private submit transactions <br>

POST <br>
localhost:9000/demoapp/v1.0/HLF/submitTransactions <br>

{
    "chaincodeName": "test-private-cc-1",
    "methodName": "saveDetails",
    "channelName": "mychannel",
    "isTransient":"true",
    "transientKey":"payload",
    "transientValue":"PAYLOAD FROM SDK",
	"inputParameters": ["org-collection","PRV000904", "{\"Name\": \"John\", \"Place\": \"India\"}"]
}

//private querying <br>
GET <br>
localhost:9000/demoapp/v1.0/HLF/getTransactions?chaincodeName=test-private-cc-1&methodName=getDetails&channelName=mychannel&value=org-collection&value=PRV000904
<br>
POST <br>
//Public submit transactions <br>
{
    "chaincodeName": "test-public-cc-1",
    "methodName": "saveDetails",
    "channelName": "mychannel",
    "isTransient":"false",
    "transientKey":"payload",
    "transientValue":"PAYLOAD FROM SDK",
	"inputParameters": ["PUB000902", "{\"Name\": \"John\", \"Place\": \"India\"}"]
}
<br>
GET  <br>
//public querying <br>
localhost:9000/demoapp/v1.0/HLF/getTransactions?chaincodeName=test-public-cc-1&methodName=getDetails&channelName=mychannel&value=PUB000902
<br>
GET <br>
//for getting block height <br>
localhost:9000/demoapp/v1.0/HLF/getBlockHeight?channelName=mychannel
<br>
POST <br>
//for subscribing events <br>
localhost:9000/demoapp/v1.0/HLF/subscribeEvent

{
    "eventNameList": ["savePrivateDetails"],
    "channelName": "mychannel",
    "chaincode": "test-private-cc-1",
    "startBlockNumber":10,
    "endBlockNumber":20
}


//Run jar file as a service
nohup java -jar HLF-Java-SDK-demo-0.0.1-SNAPSHOT.jar -Dspring.config.location=application.properties -Dserver.port=9000 &
